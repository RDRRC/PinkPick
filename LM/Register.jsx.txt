import { useEffect } from 'react';
import GuestLayout from '@/Layouts/GuestLayout';
import InputError from '@/Components/InputError';
import InputLabel from '@/Components/InputLabel';
import PrimaryButton from '@/Components/PrimaryButton';
import TextInput from '@/Components/TextInput';
import { Head, Link, useForm } from '@inertiajs/react';
export default function Register() {
    const { data, setData, post, processing, errors, reset } = useForm({
        name: '',
        email: '',
        password: '',
        password_confirmation: '',
    });
    useEffect(() =&gt; {
        return () =&gt; {
            reset('password', 'password_confirmation');
        };
    }, []);
    const submit = (e) =&gt; {
        e.preventDefault();
        post(route('register'));
    };
    return (
        &lt;GuestLayout&gt;
            &lt;Head title="註冊帳號" /&gt;
            &lt;div className="mb-6"&gt;
                &lt;Link
                    href="/"
                    className="text-sm text-gray-500 hover:text-gray-900 transition duration-150 ease-in-out"
                &gt;
                    &larr; 回商城首頁
                &lt;/Link&gt;
            &lt;/div&gt;
            &lt;form onSubmit={submit}&gt;
                &lt;div&gt;
                    &lt;InputLabel htmlFor="name" value="姓名" /&gt;
                    &lt;TextInput
                        id="name"
                        name="name"
                        value={data.name}
                        className="mt-1 block w-full"
                        autoComplete="name"
                        isFocused={true}
                        onChange={(e) =&gt; setData('name', e.target.value)}
                        required
                    /&gt;
                    &lt;InputError message={errors.name} className="mt-2" /&gt;
                &lt;/div&gt;
                &lt;div className="mt-4"&gt;
                    &lt;InputLabel htmlFor="email" value="電子郵件" /&gt;
                    &lt;TextInput
                        id="email"
                        type="email"
                        name="email"
                        value={data.email}
                        className="mt-1 block w-full"
                        autoComplete="username"
                        onChange={(e) =&gt; setData('email', e.target.value)}
                        required
                    /&gt;
                    &lt;InputError message={errors.email} className="mt-2" /&gt;
                &lt;/div&gt;
                &lt;div className="mt-4"&gt;
                    &lt;InputLabel htmlFor="password" value="密碼" /&gt;
                    &lt;TextInput
                        id="password"
                        type="password"
                        name="password"
                        value={data.password}
                        className="mt-1 block w-full"
                        autoComplete="new-password"
                        onChange={(e) =&gt; setData('password', e.target.value)}
                        required
                    /&gt;
                    &lt;InputError message={errors.password} className="mt-2" /&gt;
                &lt;/div&gt;
                &lt;div className="mt-4"&gt;
                    &lt;InputLabel htmlFor="password_confirmation" value="確認密碼" /&gt;
                    &lt;TextInput
                        id="password_confirmation"
                        type="password"
                        name="password_confirmation"
                        value={data.password_confirmation}
                        className="mt-1 block w-full"
                        autoComplete="new-password"
                        onChange={(e) =&gt; setData('password_confirmation', e.target.value)}
                        required
                    /&gt;
                    &lt;InputError message={errors.password_confirmation} className="mt-2" /&gt;
                &lt;/div&gt;
                &lt;div className="flex items-center justify-end mt-6"&gt;
                    &lt;Link
                        href={route('login')}
                        className="text-sm text-gray-600 hover:text-gray-900 rounded-md focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-pink-500 transition duration-150 ease-in-out"
                    &gt;
                        已經有帳號了？點此登入
                    &lt;/Link&gt;
                    &lt;PrimaryButton className="ml-4" disabled={processing}&gt;
                        註冊
                    &lt;/PrimaryButton&gt;
                &lt;/div&gt;
            &lt;/form&gt;
        &lt;/GuestLayout&gt;
    );
}
