import Checkbox from '@/Components/Checkbox';
import InputError from '@/Components/InputError';
import InputLabel from '@/Components/InputLabel';
import PrimaryButton from '@/Components/PrimaryButton';
import TextInput from '@/Components/TextInput';
import GuestLayout from '@/Layouts/GuestLayout';
import { Head, Link, useForm } from '@inertiajs/react';
export default function Login({ status, canResetPassword }) {
    const { data, setData, post, processing, errors, reset } = useForm({
        email: '',
        password: '',
        remember: false,
    });
    const submit = (e) =&gt; {
        e.preventDefault();
        post(route('login'), {
            onFinish: () =&gt; reset('password'),
        });
    };
    return (
        &lt;GuestLayout&gt;
            &lt;Head title="登入" /&gt;
            &lt;div className="mb-4 text-left"&gt;
                &lt;Link href={route('shop')} className="text-sm text-gray-500 hover:text-pink-600 transition"&gt;
                    ← 回商城首頁
                &lt;/Link&gt;
            &lt;/div&gt;
            {status && (
                &lt;div className="mb-4 text-sm font-medium text-green-600"&gt;
                    {status}
                &lt;/div&gt;
            )}
            &lt;form onSubmit={submit}&gt;
                &lt;div&gt;
                    &lt;InputLabel htmlFor="email" value="電子郵件" /&gt;
                    &lt;TextInput
                        id="email"
                        type="email"
                        name="email"
                        value={data.email}
                        className="mt-1 block w-full"
                        autoComplete="username"
                        isFocused={true}
                        onChange={(e) =&gt; setData('email', e.target.value)}
                    /&gt;
                    &lt;InputError message={errors.email} className="mt-2" /&gt;
                &lt;/div&gt;
                &lt;div className="mt-4"&gt;
                    &lt;InputLabel htmlFor="password" value="密碼" /&gt;
                    &lt;TextInput
                        id="password"
                        type="password"
                        name="password"
                        value={data.password}
                        className="mt-1 block w-full"
                        autoComplete="current-password"
                        onChange={(e) =&gt; setData('password', e.target.value)}
                    /&gt;
                    &lt;InputError message={errors.password} className="mt-2" /&gt;
                &lt;/div&gt;
                &lt;div className="mt-4 block"&gt;
                    &lt;label className="flex items-center"&gt;
                        &lt;Checkbox
                            name="remember"
                            checked={data.remember}
                            onChange={(e) =&gt;
                                setData('remember', e.target.checked)
                            }
                        /&gt;
                        &lt;span className="ms-2 text-sm text-gray-600"&gt;
                            記住我
                        &lt;/span&gt;
                    &lt;/label&gt;
                &lt;/div&gt;
                &lt;div className="mt-6 flex flex-wrap items-center justify-end gap-4"&gt;
                    {canResetPassword && (
                        &lt;Link
                            href={route('password.request')}
                            className="rounded-md text-sm text-gray-600 underline hover:text-gray-900 focus:outline-none focus:ring-2 focus:ring-pink-500 focus:ring-offset-2"
                        &gt;
                            忘記密碼？
                        &lt;/Link&gt;
                    )}
                    &lt;Link
                        href={route('register')}
                        className="rounded-md text-sm text-pink-600 underline hover:text-pink-800 focus:outline-none focus:ring-2 focus:ring-pink-500 focus:ring-offset-2 font-medium"
                    &gt;
                        還沒有帳號？點此註冊
                    &lt;/Link&gt;
                    &lt;PrimaryButton disabled={processing}&gt;
                        登入
                    &lt;/PrimaryButton&gt;
                &lt;/div&gt;
            &lt;/form&gt;
        &lt;/GuestLayout&gt;
    );
}
