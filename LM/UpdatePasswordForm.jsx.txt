import { useRef } from 'react';
import InputError from '@/Components/InputError';
import InputLabel from '@/Components/InputLabel';
import PrimaryButton from '@/Components/PrimaryButton';
import TextInput from '@/Components/TextInput';
import { useForm } from '@inertiajs/react';
import { Transition } from '@headlessui/react';
export default function UpdatePasswordForm({ className = '' }) {
    const passwordInput = useRef();
    const currentPasswordInput = useRef();
    const { data, setData, errors, put, reset, processing, recentlySuccessful } = useForm({
        current_password: '',
        password: '',
        password_confirmation: '',
    });
    const updatePassword = (e) =&gt; {
        e.preventDefault();
        put(route('password.update'), {
            preserveScroll: true,
            onSuccess: () =&gt; reset(),
            onError: (errors) =&gt; {
                if (errors.password) {
                    reset('password', 'password_confirmation');
                    passwordInput.current.focus();
                }
                if (errors.current_password) {
                    reset('current_password');
                    currentPasswordInput.current.focus();
                }
            },
        });
    };
    return (
        &lt;section className={className}&gt;
            &lt;header&gt;
                &lt;h2 className="text-lg font-medium text-gray-900"&gt;修改密碼&lt;/h2&gt;
                &lt;p className="mt-1 text-sm text-gray-600"&gt;
                    請確保您的帳號使用長度足夠且隨機的密碼，以保障帳號安全。
                &lt;/p&gt;
            &lt;/header&gt;
            &lt;form onSubmit={updatePassword} className="mt-6 space-y-6"&gt;
                &lt;div&gt;
                    &lt;InputLabel htmlFor="current_password" value="目前密碼" /&gt;
                    &lt;TextInput
                        id="current_password"
                        ref={currentPasswordInput}
                        value={data.current_password}
                        onChange={(e) =&gt; setData('current_password', e.target.value)}
                        type="password"
                        className="mt-1 block w-full"
                        autoComplete="current-password"
                    /&gt;
                    &lt;InputError message={errors.current_password} className="mt-2" /&gt;
                &lt;/div&gt;
                &lt;div&gt;
                    &lt;InputLabel htmlFor="password" value="新密碼" /&gt;
                    &lt;TextInput
                        id="password"
                        ref={passwordInput}
                        value={data.password}
                        onChange={(e) =&gt; setData('password', e.target.value)}
                        type="password"
                        className="mt-1 block w-full"
                        autoComplete="new-password"
                    /&gt;
                    &lt;InputError message={errors.password} className="mt-2" /&gt;
                &lt;/div&gt;
                &lt;div&gt;
                    &lt;InputLabel htmlFor="password_confirmation" value="確認新密碼" /&gt;
                    &lt;TextInput
                        id="password_confirmation"
                        value={data.password_confirmation}
                        onChange={(e) =&gt; setData('password_confirmation', e.target.value)}
                        type="password"
                        className="mt-1 block w-full"
                        autoComplete="new-password"
                    /&gt;
                    &lt;InputError message={errors.password_confirmation} className="mt-2" /&gt;
                &lt;/div&gt;
                &lt;div className="flex items-center gap-4"&gt;
                    &lt;PrimaryButton disabled={processing}&gt;儲存設定&lt;/PrimaryButton&gt;
                    &lt;Transition
                        show={recentlySuccessful}
                        enter="transition ease-in-out"
                        enterFrom="opacity-0"
                        leave="transition ease-in-out"
                        leaveTo="opacity-0"
                    &gt;
                        &lt;p className="text-sm text-green-600 font-medium"&gt;儲存成功。&lt;/p&gt;
                    &lt;/Transition&gt;
                &lt;/div&gt;
            &lt;/form&gt;
        &lt;/section&gt;
    );
}
