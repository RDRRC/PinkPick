import GuestLayout from '@/Layouts/GuestLayout';
import InputError from '@/Components/InputError';
import PrimaryButton from '@/Components/PrimaryButton';
import TextInput from '@/Components/TextInput';
import { Head, Link, useForm } from '@inertiajs/react';
export default function ForgotPassword({ status }) {
    const { data, setData, post, processing, errors } = useForm({
        email: '',
    });
    const submit = (e) =&gt; {
        e.preventDefault();
        post(route('password.email'));
    };
    return (
        &lt;GuestLayout&gt;
            &lt;Head title="忘記密碼" /&gt;
            &lt;div className="mb-6"&gt;
                &lt;Link
                    href="/"
                    className="text-sm text-gray-500 hover:text-gray-900 transition duration-150 ease-in-out"
                &gt;
                    &larr; 回商城首頁
                &lt;/Link&gt;
            &lt;/div&gt;
            &lt;div className="mb-4 text-sm text-gray-600"&gt;
                忘記密碼了嗎？沒問題。請輸入您的電子郵件地址，我們將發送一封密碼重設連結給您，讓您重新設定密碼。
            &lt;/div&gt;
            {status && &lt;div className="mb-4 font-medium text-sm text-green-600"&gt;{status}&lt;/div&gt;}
            &lt;form onSubmit={submit}&gt;
                &lt;TextInput
                    id="email"
                    type="email"
                    name="email"
                    value={data.email}
                    className="mt-1 block w-full"
                    isFocused={true}
                    onChange={(e) =&gt; setData('email', e.target.value)}
                    placeholder="請輸入您的電子郵件"
                /&gt;
                &lt;InputError message={errors.email} className="mt-2" /&gt;
                &lt;div className="flex items-center justify-end mt-6"&gt;
                    &lt;PrimaryButton disabled={processing}&gt;
                        發送密碼重設連結
                    &lt;/PrimaryButton&gt;
                &lt;/div&gt;
            &lt;/form&gt;
        &lt;/GuestLayout&gt;
    );
}
