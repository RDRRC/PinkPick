namespace App\Http\Controllers;
abstract class Controller
{
    public function sendResponse($payload, $message = '操作成功')
    {
        return response()->json([
            'success' => true,
            'payload' => $payload,
            'message' => $message,
        ], 200);
    }
    public function sendError($errorMessages, $code = 404, $errorCode = 'UNKNOWN_ERROR')
    {
        return response()->json([
            'success'   => false,
            'message'   => $errorMessages,  
            'errorCode' => $errorCode,      
        ], $code);
    }
}
